53 Convergence_Data solve(const Matrix_Sparse& a_matrix, Vector_Dense<Scalar, 0>& x_vector,
54 const Vector_Dense<Scalar, 0>& b_vector) {
55
56 switch(solver.index()) {
57 case 0:
58 ERROR("LU solver does not support sparse matrices.");
59 case 1:
60 ERROR("LUP solver does not support sparse matrices.");
61 case 2:
62 return std::get<std::unique_ptr<Solver_Jacobi>>(solver)->solve_system(a_matrix, x_vector, b_vector);
63 case 3:
64 return std::get<std::unique_ptr<Solver_Gauss_Seidel>>(solver)->solve_system(a_matrix, x_vector, b_vector);
65 case 4:
66 return std::get<std::unique_ptr<Sover_Sor>>(solver)->solve_system(a_matrix, x_vector, b_vector);
67 default:
68 ERROR("Unknown or uninitialized solver.");
69 exit(1);
70 }
71 };
72
73 Convergence_Data solve(Vector_Dense<Scalar, 0>& x_vector, const Vector_Dense<Scalar, 0>& b_vector) {
74
75 switch(solver.index()) {
76 case 0:
77 return std::get<std::unique_ptr<Solver_LU<0>>>(solver)->solve_system(x_vector, b_vector);
78 case 1:
79 return std::get<std::unique_ptr<Solver_LUP<0>>>(solver)->solve_system(x_vector, b_vector);
80 case 2:
81 ERROR("Jacobi solver does not support sparse matrices.");
82 case 3:
83 ERROR("Gauss Seidel solver does not support sparse matrices.");
84 case 4:
85 ERROR("SOR solver does not support sparse matrices.");
86 default:
87 ERROR("Unknown or uninitialized solver.");
88 exit(1);
89 }
90 };
