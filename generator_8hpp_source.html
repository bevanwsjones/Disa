81template<bool _directed>
82Adjacency_Graph<_directed> create_graph_structured(std::size_t number_vertices) {
83 Adjacency_Graph<_directed> structured;
84 for(std::size_t i_y = 0; i_y < number_vertices; ++i_y) {
85 for(std::size_t i_x = 0; i_x < number_vertices; ++i_x) {
86 const std::size_t i_vertex = i_y * (number_vertices) + i_x;
87 if(!_directed && i_x != 0) structured.insert({i_vertex - 1, i_vertex});
88 if(i_x != number_vertices - 1) structured.insert({i_vertex, i_vertex + 1});
89 if(!_directed && i_y != 0) structured.insert({i_vertex - number_vertices, i_vertex});
90 if(i_y < number_vertices - 1) structured.insert({i_vertex, i_vertex + number_vertices});
91 }
92 }
93 return structured;
94}
