54inline std::basic_string<char> console_format(const Log_Level level, const std::source_location& location) {
55 auto const iter = std::basic_string_view<char>(location.file_name()).find_last_of('/');
56 const auto suffix = std::basic_string<char>(location.file_name()).substr(iter + 1) + "::"
57 + std::to_string(location.line()) + "]: ";
58 switch (level) {
59 case Log_Level::Debug:
60 return "[Debug|" + std::basic_string(suffix);
61 case Log_Level::Info:
62 return "[Info |" + std::basic_string(suffix);
63 case Log_Level::Warning:
64 return "[Warn |" + std::basic_string(suffix);
65 case Log_Level::Error:
66 return "[Error|" + std::basic_string(suffix);
67 default:
68 std::cout<<"[Error|" + std::basic_string(suffix) + " Unrecognised log level parsed.";
69 exit(1);
70 }
71}//console_format
