56 Convergence_Data solve(const Matrix_Sparse& a_matrix, Vector_Dense<0>& x_vector,
57 const Vector_Dense<0>& b_vector) {
58
59 switch(solver.index()) {
60 case 0: ERROR("LU solver does not support sparse matrices.");
61 case 1: ERROR("LUP solver does not support sparse matrices.");
62 case 2: return std::get<std::unique_ptr<Solver_Jacobi> >(solver)->solve_system(a_matrix, x_vector, b_vector);
63 case 3: return std::get<std::unique_ptr<Solver_Gauss_Seidel> >(solver)->solve_system(a_matrix, x_vector, b_vector);
64 case 4: return std::get<std::unique_ptr<Sover_Sor> >(solver)->solve_system(a_matrix, x_vector, b_vector);
65 default:
66 ERROR("Unknown or uninitialized solver.");
67 exit(1);
68 }
69 };
70
71 Convergence_Data solve(Vector_Dense<0>& x_vector, const Vector_Dense<0>& b_vector) {
72
73 switch(solver.index()) {
74 case 0: return std::get<std::unique_ptr<Solver_LU<0> > >(solver)->solve_system(x_vector, b_vector);
75 case 1: return std::get<std::unique_ptr<Solver_LUP<0> > >(solver)->solve_system(x_vector, b_vector);
76 case 2: ERROR("Jacobi solver does not support sparse matrices.");
77 case 3: ERROR("Gauss Seidel solver does not support sparse matrices.");
78 case 4: ERROR("SOR solver does not support sparse matrices.");
79 default:
80 ERROR("Unknown or uninitialized solver.");
81 exit(1);
82 }
83 };
