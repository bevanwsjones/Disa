564
573 template<typename _matrix = _matrix_type, typename std::enable_if_t<!std::is_const_v<_matrix>, bool> = true>
574 constexpr Scalar& operator[](const std::size_t& i_column) {
575 auto iter_element = matrix->lower_bound(row_index, i_column);
576 if(iter_element == end() || i_column != iter_element.i_column())
577 iter_element = matrix->insert(row_index, i_column, Scalar()).first;
578 return *iter_element;
579 };
580
588 constexpr const Scalar& operator[](const std::size_t& i_column) const {
589 ASSERT_DEBUG(matrix->contains(row_index, i_column),
590 "Trying to access a zero element at [" + matrix->row_column(row_index, i_column) + "].");
591 return *matrix->find(row_index, i_column);
592 };

1103template<std::size_t _size>
1104Vector_Dense<_size> operator*(const Matrix_Sparse& matrix, const Vector_Dense<_size>& vector) {
1105 ASSERT_DEBUG(matrix.size_column() == vector.size(),
1106 "Incompatible vector-matrix dimensions, " + std::to_string(matrix.size_row()) + "," +
1107 std::to_string(matrix.size_column()) + " vs. " + std::to_string(vector.size()) + ".");
1108 ASSERT_DEBUG(!_size || matrix.size_row() == vector.size(), "For static vectors the matrix must be square.");
1109 return Vector_Dense<_size>([&](const std::size_t i_row, Scalar value = 0) {
1110 FOR_ITER(iter, *(matrix.begin() + i_row)) value += *iter*vector[iter.i_column()];
1111 return value; }, matrix.size_row());
