62  template<class _matrix_type, class _vector_type>
63  void construct_laplace_2d(_matrix_type& a_matrix, _vector_type& b_vector, _vector_type& x_vector) {
64 
65  ASSERT_DEBUG(a_matrix.size_row() == a_matrix.size_column(), "Matrix must be square.");
66  ASSERT_DEBUG(a_matrix.size_row() == b_vector.size(), "Matrix-vector size mismatch, b-vector.");
67  ASSERT_DEBUG(a_matrix.size_row() == x_vector.size(), "Matrix-vector size mismatch, x-vector.");
68  const int number_of_nodes = static_cast<int>(b_vector.size());
69  ASSERT_DEBUG(number_of_nodes >= 9, "Must be greater equal than 9.");
70  const int x_nodes = sqrt(number_of_nodes);
71 
72  a_matrix *= 0.0;
73  const Disa::Scalar delta_x = static_cast<Disa::Scalar>(1.0/(x_nodes - 1.0));
74  const Disa::Scalar delta_x_squared = 1.0/(delta_x*delta_x);
75 
76  // populate
77  FOR(i_node, number_of_nodes)
78  {
79  b_vector[i_node] = source(i_node, x_nodes);
80  x_vector[i_node] = analytical_solution(i_node, x_nodes);
81 
82  if((i_node) % x_nodes == 0 || (i_node + 1) % x_nodes == 0 || i_node < x_nodes || i_node >= (number_of_nodes - x_nodes)) {
83  FOR_EACH_REF(column, a_matrix[i_node]) column = 0.0;
84  a_matrix[i_node][i_node] = 1.0;
85  b_vector[i_node] = analytical_solution(i_node, x_nodes);
86  }
87  else {
88  if((i_node + x_nodes) < number_of_nodes) a_matrix[i_node][i_node + x_nodes] = -1.0*delta_x_squared;
89  if(i_node%x_nodes != 0) a_matrix[i_node][i_node - 1] = -1.0*delta_x_squared;
90  a_matrix[i_node][i_node] = 4*delta_x_squared;
91  if((i_node + 1)%x_nodes != 0) a_matrix[i_node][i_node + 1] = -1.0*delta_x_squared;
92  if((i_node - x_nodes) >= 0) a_matrix[i_node][i_node - x_nodes] = -1.0*delta_x_squared;
93  }
94 
95  }
96  }

99  const Disa::Scalar delta_x = 1.0/static_cast<Disa::Scalar>(size_x - 1.0);
100  return {static_cast<Disa::Scalar>((i_node % size_x)*delta_x),
101  static_cast<Disa::Scalar>(static_cast<int>(i_node / size_x))*delta_x};

111  Disa::Scalar source(const std::size_t i_node, const std::size_t size_x) {
112  // -2 [(1 - 6x^2)y^2(y^2 - 1) + (6y^2 - 1)x^2(1 - x^2)]
113  const auto [co_ordinate_x, co_ordinate_y] = co_ordinate(i_node, size_x);
114  const Disa::Scalar squared_x = co_ordinate_x*co_ordinate_x;
115  const Disa::Scalar squared_y = co_ordinate_y*co_ordinate_y;
116  const Disa::Scalar value = -2.0*((1.0 - 6.0*squared_x)*squared_y*(squared_y - 1.0)
117  + (6.0*squared_y - 1.0)*squared_x*(1.0 - squared_x));
118  return value;
119  }

128  Disa::Scalar analytical_solution(const std::size_t i_node, const std::size_t size_x) {
129  const auto [co_ordinate_x, co_ordinate_y] = co_ordinate(i_node, size_x);
130  const Disa::Scalar squared_x = co_ordinate_x*co_ordinate_x;
131  const Disa::Scalar squared_y = co_ordinate_y*co_ordinate_y;
132  return (squared_x - squared_x*squared_x)*(squared_y*squared_y - squared_y);
133  }
