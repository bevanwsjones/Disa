147struct Convergence_Criteria {
148
149 std::size_t min_iterations {0};
150 std::size_t max_iteration {std::numeric_limits<std::size_t>::max()};
151 Scalar tolerance {scalar_max};
152
158 [[nodiscard]] inline bool is_converged(const Convergence_Data& data) const {
159 if(data.iteration < min_iterations) return false;
160 if(data.iteration > max_iteration) return true;
161 if(data.residual_normalised > tolerance) return false;
162 if(data.residual_max_normalised > 10.0*tolerance) return false;
163 return true;
164 }
165};
