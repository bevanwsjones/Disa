45 template<Solver_Type _solver_type, std::size_t _size, bool _pivot>
46 bool Direct_Lower_Upper_Factorisation<_solver_type, _size, _pivot>::factorise(
47  const Matrix_Dense<Scalar, _size, _size>& a_matrix) {
48 
49  // Initialise factorisation data.
50  factorised = false;
51  lu_factorised = a_matrix;
52  if(_pivot){
53  pivots.resize(lu_factorised.size_column());
54  std::iota(pivots.begin(), pivots.end(), 0);
55  }
56 
57  // Factorise A into L and U.
58  for(std::size_t i_row = 0; i_row < lu_factorised.size_row(); ++i_row) {
59 
60  // Pivoting
61  if(_pivot) {
62 
63  // Find largest remaining column value to pivot on.
64  Scalar max = 0.0;
65  std::size_t i_max = i_row;
66  for(std::size_t i_row_sweep = i_row; i_row_sweep < lu_factorised.size_column(); ++i_row_sweep) {
67  const Scalar absA = std::abs(lu_factorised[i_row_sweep][i_row]);
68  if(is_nearly_greater(absA, max)) {
69  max = absA;
70  i_max = i_row_sweep;
71  }
72  }
73 
74  // Pivot if larger value found.
75  if (i_max != i_row) {
76  std::swap(pivots[i_row], pivots[i_max]);
77  std::swap(lu_factorised[i_row], lu_factorised[i_max]);
78  }
79  }
80 
81  // Check degeneracy.
82  if(std::abs(lu_factorised[i_row][i_row]) < factorisation_tolerance) return false;
83 
84  // Decomposition, actual factorisation to compute L and U.
85  for(std::size_t i_row_sweep = i_row + 1; i_row_sweep < lu_factorised.size_row(); ++i_row_sweep) {
86  lu_factorised[i_row_sweep][i_row] /= lu_factorised[i_row][i_row];
87 
88  for (std::size_t i_column_sweep = i_row + 1; i_column_sweep < lu_factorised.size_column(); ++i_column_sweep)
89  lu_factorised[i_row_sweep][i_column_sweep] -=
90  lu_factorised[i_row_sweep][i_row] * lu_factorised[i_row][i_column_sweep];
91  }
92  }
93 
94  factorised = true;
95  return true;
96 }

106 template<Solver_Type _solver_type, std::size_t _size, bool _pivot>
107 Convergence_Data Direct_Lower_Upper_Factorisation<_solver_type, _size, _pivot>::solve_system(
108  Vector_Dense<Scalar, _size>& x_vector, const Vector_Dense<Scalar, _size>& b_vector) {
109 
110  ASSERT_DEBUG(b_vector.size() == lu_factorised.size_row(), "Constant vector not of the correct size.");
111 
112  Convergence_Data convergence_data = Convergence_Data();
113  if(!factorised) return convergence_data;
114 
115  ++convergence_data.iteration;
116  x_vector.resize(b_vector.size());
117  for(std::size_t i_row = 0; i_row < lu_factorised.size_row(); ++i_row) {
118  x_vector[i_row] = b_vector[_pivot ? pivots[i_row] : i_row];
119 
120  for(std::size_t i_column = 0; i_column < i_row; ++i_column)
121  x_vector[i_row] -= lu_factorised[i_row][i_column] * x_vector[i_column];
122  }
123 
124  for(std::size_t i_row = lu_factorised.size_row() - 1; i_row != std::numeric_limits<std::size_t>::max(); --i_row) {
125  for(std::size_t i_column = i_row + 1; i_column < lu_factorised.size_column(); ++i_column)
126  x_vector[i_row] -= lu_factorised[i_row][i_column] * x_vector[i_column];
127  x_vector[i_row] /= lu_factorised[i_row][i_row];
128  }
129 
130  convergence_data.converged = true;
131  return convergence_data;
132 }
