142template<Solver_Type _solver_type, std::size_t _size, bool _pivot>
143bool Direct_Lower_Upper_Factorisation<_solver_type, _size, _pivot>::factorise(
144const Matrix_Dense<Scalar, _size, _size>& a_matrix) {
145
146 // Initialise factorisation data.
147 factorised = false;
148 lu_factorised = a_matrix;
149 if(_pivot) {
150 pivots.resize(lu_factorised.size_column());
151 std::iota(pivots.begin(), pivots.end(), 0);
152 }
153
154 // Factorise A into L and U.
155 for(std::size_t i_row = 0; i_row < lu_factorised.size_row(); ++i_row) {
156
157 // Pivoting
158 if(_pivot) {
159
160 // Find largest remaining column value to pivot on.
161 Scalar max = 0.0;
162 std::size_t i_max = i_row;
163 for(std::size_t i_row_sweep = i_row; i_row_sweep < lu_factorised.size_column(); ++i_row_sweep) {
164 const Scalar absA = std::abs(lu_factorised[i_row_sweep][i_row]);
165 if(is_nearly_greater(absA, max)) {
166 max = absA;
167 i_max = i_row_sweep;
168 }
169 }
170
171 // Pivot if larger value found.
172 if(i_max != i_row) {
173 std::swap(pivots[i_row], pivots[i_max]);
174 std::swap(lu_factorised[i_row], lu_factorised[i_max]);
175 }
176 }
177
178 // Check degeneracy.
179 if(std::abs(lu_factorised[i_row][i_row]) < factorisation_tolerance) return false;
180
181 // Decomposition, actual factorisation to compute L and U.
182 for(std::size_t i_row_sweep = i_row + 1; i_row_sweep < lu_factorised.size_row(); ++i_row_sweep) {
183 lu_factorised[i_row_sweep][i_row] /= lu_factorised[i_row][i_row];
184
185 for(std::size_t i_column_sweep = i_row + 1; i_column_sweep < lu_factorised.size_column(); ++i_column_sweep)
186 lu_factorised[i_row_sweep][i_column_sweep] -=
187 lu_factorised[i_row_sweep][i_row] * lu_factorised[i_row][i_column_sweep];
188 }
189 }
190
191 factorised = true;
192 return true;
193}

203template<Solver_Type _solver_type, std::size_t _size, bool _pivot>
204Convergence_Data Direct_Lower_Upper_Factorisation<_solver_type, _size, _pivot>::solve_system(
205Vector_Dense<Scalar, _size>& x_vector, const Vector_Dense<Scalar, _size>& b_vector) {
206
207 ASSERT_DEBUG(b_vector.size() == lu_factorised.size_row(), "Constant vector not of the correct size.");
208
209 Convergence_Data convergence_data = Convergence_Data();
210 if(!factorised) return convergence_data;
211
212 ++convergence_data.iteration;
213 x_vector.resize(b_vector.size());
214 for(std::size_t i_row = 0; i_row < lu_factorised.size_row(); ++i_row) {
215 x_vector[i_row] = b_vector[_pivot ? pivots[i_row] : i_row];
216
217 for(std::size_t i_column = 0; i_column < i_row; ++i_column)
218 x_vector[i_row] -= lu_factorised[i_row][i_column] * x_vector[i_column];
219 }
220
221 for(std::size_t i_row = lu_factorised.size_row() - 1; i_row != std::numeric_limits<std::size_t>::max(); --i_row) {
222 for(std::size_t i_column = i_row + 1; i_column < lu_factorised.size_column(); ++i_column)
223 x_vector[i_row] -= lu_factorised[i_row][i_column] * x_vector[i_column];
224 x_vector[i_row] /= lu_factorised[i_row][i_row];
225 }
226
227 convergence_data.converged = true;
228 return convergence_data;
229}
