60struct Matrix_Dense : public std::array<Vector_Dense<_type, _col>, _row> {

63 static constexpr std::size_t row = _row;

92 explicit Matrix_Dense(const std::function<Scalar(std::size_t, std::size_t)>& lambda, std::size_t row = _row,

157 template<std::size_t _row_other, std::size_t _col_other>
158 constexpr matrix_type& operator+=(const Matrix_Dense<_type, _row_other, _col_other>& matrix) {
159 ASSERT_DEBUG(size() == matrix.size(), "Incompatible matrix dimensions, " + std::to_string(size_row()) + "," +

191 template<std::size_t _row_other, std::size_t _col_other>
192 constexpr matrix_type& operator*=(const Matrix_Dense<_type, _row_other, _col_other>& matrix) {
193 // For static containers the matrices must be square.
194 ASSERT_DEBUG(size() == matrix.size(), "Incompatible matrix dimensions, " + std::to_string(size_row()) + "," +
195 std::to_string(size_column()) + " vs. " + std::to_string(matrix.size_row()) +
196 "," + std::to_string(matrix.size_column()) + ".");
197 matrix_type copy;
198 std::swap(*this, copy);
199 FOR(i_row, size_row()) {
200 FOR(i_row_other, matrix.size_row()) {
201 FOR(i_column, size_column()) {
202 if(i_row_other == 0) (*this)[i_row][i_column] = 0.0;
203 (*this)[i_row][i_column] += copy[i_row][i_row_other] * matrix[i_row_other][i_column];
204 }
205 }
206 }
207 return *this;
208 }

404template<class matrix_0, class matrix_1>
405struct Matrix_Static_Demoter {
406 static constexpr bool is_dynamic = matrix_0::is_dynamic || matrix_1::is_dynamic;
407 static constexpr std::size_t row_new = is_dynamic ? 0 : matrix_0::row;
408 static constexpr std::size_t col_new = is_dynamic ? 0 : matrix_1::col;
409 using type =
410 Matrix_Dense<typename matrix_0::value_type, row_new,
411 col_new>;
412};

459 ASSERT_DEBUG(matrix.size_column() == vector.size(),
