196  std::swap(*this, copy);

200  if (i_row_other == 0) (*this)[i_row][i_column] = 0.0;
201  (*this)[i_row][i_column] += copy[i_row][i_row_other]*matrix[i_row_other][i_column];
202  }
203  }
204  }
