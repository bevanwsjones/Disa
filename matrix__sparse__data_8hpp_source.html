155 template<typename _value_type, typename _index_type, typename _arg_index_type>
156 std::enable_if<std::is_convertible_v<_arg_index_type, _index_type>, void>::type resize(
157 CSR_Data<_value_type, _index_type>& data, const _arg_index_type& row, const _arg_index_type& column) {
158 
159  ASSERT_DEBUG(row >= 0, "New row size must be positive.");
160  ASSERT_DEBUG(column >= 0, "New columns size must be positive.");
161 
162  // Set up alias.
163  using s_index_type = std::make_signed_t<_index_type>;
164  auto& row_offset = data.row_offset;
165  auto& column_index = data.column_index;
166  auto& value = data.value;
167 
168  // resize rows first
169  if(row < size_row(data)) {
170  column_index.resize(row_offset[row]);
171  value.resize(row_offset[row]);
172  }
173  row_offset.resize(row + 1, row_offset.empty() ? 0 : row_offset.back());
174 
175  // resize columns
176  // Note: Since we may have to do column erases over multiple rows we do it directly here rather than use erase().
177  if(column < size_column(data)) {
178  _index_type offset_loss = 0;
179  for(_index_type row = 0; row < size_row(data); ++row) {
180  const auto& iter_column_end = column_index.begin() + static_cast<s_index_type>(row_offset[row + 1] - offset_loss);
181  auto iter_column =
182  std::upper_bound(column_index.begin() + static_cast<s_index_type>(row_offset[row]), iter_column_end, column - 1);
183  if(iter_column != iter_column_end) {
184  const auto& start_distance = std::distance(column_index.begin(), iter_column);
185  const auto& end_distance = std::distance(column_index.begin(), iter_column_end);
186  column_index.erase(iter_column, iter_column_end);
187  value.erase(value.begin() + start_distance, value.begin() + end_distance);
188  offset_loss += end_distance - start_distance;
189  }
190  row_offset[row + 1] -= offset_loss;
191  }
192  }
193  data.columns = column;
194 }

214 template<typename _value_type, typename _index_type, typename _arg_index_type>
215 std::pair<typename CSR_Data<_value_type, _index_type>::iterator, bool> insert(CSR_Data<_value_type, _index_type>& data,
216  const _arg_index_type& row,
217  const _arg_index_type& column,
218  const _value_type& value) {
219 
220  // Check sizes
221  ASSERT_DEBUG(row >= 0, "Row index must be positive.");
222  ASSERT_DEBUG(column >= 0, "Columns index must be positive.");
223 
224  // Resize if we need to.
225  if(row >= size_row(data)) resize(data, row + 1, static_cast<_arg_index_type>(size_column(data)));
226  if(column >= size_column(data)) resize(data, static_cast<_arg_index_type>(size_row(data)), column + 1);
227 
228  // Search for new insert
229  const auto iter_row_start = data.row_offset.begin() + row; // tricky, only create after resizing.
230  auto [iter_row, iter_column, iter_value] = lower_bound(data, row, column);
231 
232  // Entry exists, return no insert.
233  if(iter_row == iter_row_start && *iter_column == column)
234  return {std::make_tuple(iter_row, iter_column, iter_value), false};
235 
236  // Insert new entry.
237  for(auto non_zeros = iter_row_start + 1; non_zeros < data.row_offset.end(); ++(*non_zeros++)) {}; // increment and ++
238  iter_column = data.column_index.insert(iter_column, column);
239  iter_value = data.value.insert(iter_value, value);
240  return {std::make_tuple(iter_row_start, iter_column, iter_value), true};
241 }

262 template<typename _value_type, typename _index_type, typename _arg_index_type>
263 std::enable_if<std::is_convertible_v<_arg_index_type, _index_type>,
264  typename CSR_Data<_value_type, _index_type>::const_iterator>::type
265 lower_bound(const CSR_Data<_value_type, _index_type>& data, const _arg_index_type& row, const _arg_index_type& column) {
266  // Check sizes
267  ASSERT_DEBUG(row >= 0, "Row index must be positive.");
268  ASSERT_DEBUG(column >= 0, "Columns index must be positive.");
269 
270  if(row < size_row(data)) {
271  const auto& iter_row = data.row_offset.cbegin() + row;
272  const auto& iter_start = data.column_index.cbegin() + data.row_offset[row];
273  const auto& iter_end = data.column_index.cbegin() + data.row_offset[row + 1];
274  const auto& iter_lower = std::lower_bound(iter_start, iter_end, column);
275  return std::make_tuple(iter_lower != iter_end ? iter_row : (iter_row + 1), iter_lower,
276  data.value.cbegin() + std::distance(data.column_index.cbegin(), iter_lower));
277  } else return std::make_tuple(data.row_offset.cend(), data.column_index.cend(), data.value.cend());
278 }
