153template<typename _value_type, typename _index_type, typename _arg_index_type>
154std::enable_if<std::is_convertible_v<_arg_index_type, _index_type>, void>::type resize(
155CSR_Data<_value_type, _index_type>& data, const _arg_index_type& row, const _arg_index_type& column) {
156
157 ASSERT_DEBUG(row >= 0, "New row size must be positive.");
158 ASSERT_DEBUG(column >= 0, "New columns size must be positive.");
159
160 // Set up alias.
161 using s_index_type = std::make_signed_t<_index_type>;
162 auto& row_offset = data.row_offset;
163 auto& i_column = data.i_column;
164 auto& value = data.value;
165
166 // resize rows first
167 if(row < size_row(data)) {
168 i_column.resize(row_offset[row]);
169 value.resize(row_offset[row]);
170 }
171 row_offset.resize(row + 1, row_offset.empty() ? 0 : row_offset.back());
172
173 // resize columns
174 // Note: Since we may have to do column erases over multiple rows we do it directly here rather than use erase().
175 if(column < size_column(data)) {
176 _index_type offset_loss = 0;
177 for(_index_type row = 0; row < size_row(data); ++row) {
178 const auto& iter_column_end = i_column.begin() + static_cast<s_index_type>(row_offset[row + 1] - offset_loss);
179 auto iter_column =
180 std::upper_bound(i_column.begin() + static_cast<s_index_type>(row_offset[row]), iter_column_end, column - 1);
181 if(iter_column != iter_column_end) {
182 const auto& start_distance = std::distance(i_column.begin(), iter_column);
183 const auto& end_distance = std::distance(i_column.begin(), iter_column_end);
184 i_column.erase(iter_column, iter_column_end);
185 value.erase(value.begin() + start_distance, value.begin() + end_distance);
186 offset_loss += end_distance - start_distance;
187 }
188 row_offset[row + 1] -= offset_loss;
189 }
190 }
191 data.columns = column;
192}

211template<typename _value_type, typename _index_type, typename _arg_index_type>
212std::pair<typename CSR_Data<_value_type, _index_type>::iterator, bool> insert(CSR_Data<_value_type, _index_type>& data,
213 const _arg_index_type& row,
214 const _arg_index_type& column,
215 const _value_type& value) {
216
217 // Check sizes
218 ASSERT_DEBUG(row >= 0, "Row index must be positive.");
219 ASSERT_DEBUG(column >= 0, "Columns index must be positive.");
220
221 // Resize if we need to.
222 if(row >= size_row(data)) resize(data, row + 1, static_cast<_arg_index_type>(size_column(data)));
223 if(column >= size_column(data)) resize(data, static_cast<_arg_index_type>(size_row(data)), column + 1);
224
225 // Search for new insert
226 const auto iter_row_start = data.row_offset.begin() + row; // tricky, only create after resizing.
227 auto [iter_row, iter_column, iter_value] = lower_bound(data, row, column);
228
229 // Entry exists, return no insert.
230 if(iter_row == iter_row_start && *iter_column == column)
231 return {std::make_tuple(iter_row, iter_column, iter_value), false};
232
233 // Insert new entry.
234 for(auto non_zeros = iter_row_start + 1; non_zeros < data.row_offset.end(); ++(*non_zeros++)) {}; // increment and ++
235 iter_column = data.i_column.insert(iter_column, column);
236 iter_value = data.value.insert(iter_value, value);
237 return {std::make_tuple(iter_row_start, iter_column, iter_value), true};
238}

256template<typename _value_type, typename _index_type, typename _arg_index_type>
257std::pair<typename CSR_Data<_value_type, _index_type>::iterator, bool> insert_or_assign(
258CSR_Data<_value_type, _index_type>& data, const _arg_index_type& row, const _arg_index_type& column,
259const _value_type& value) {
260 auto [iter, is_insert] = insert(data, row, column, value);
261 if(!is_insert) *std::get<2>(iter) = value;
262 return {iter, is_insert};
263}

282template<typename _value_type, typename _index_type, typename _arg_index_type>
283std::enable_if<std::is_convertible_v<_arg_index_type, _index_type>,
284 typename CSR_Data<_value_type, _index_type>::const_iterator>::type
285lower_bound(const CSR_Data<_value_type, _index_type>& data, const _arg_index_type& row, const _arg_index_type& column) {
286 // Check sizes
287 ASSERT_DEBUG(row >= 0, "Row index must be positive.");
288 ASSERT_DEBUG(column >= 0, "Columns index must be positive.");
289
290 if(row < size_row(data)) {
291 const auto& iter_row = data.row_offset.cbegin() + row;
292 const auto& iter_start = data.i_column.cbegin() + data.row_offset[row];
293 const auto& iter_end = data.i_column.cbegin() + data.row_offset[row + 1];
294 const auto& iter_lower = std::lower_bound(iter_start, iter_end, column);
295 return std::make_tuple(iter_lower != iter_end ? iter_row : (iter_row + 1), iter_lower,
296 data.value.cbegin() + std::distance(data.i_column.cbegin(), iter_lower));
297 } else return std::make_tuple(data.row_offset.cend(), data.i_column.cend(), data.value.cend());
298}

317template<typename _value_type, typename _index_type, typename _arg_index_type>
318std::enable_if<std::is_convertible_v<_arg_index_type, _index_type>,
319 typename CSR_Data<_value_type, _index_type>::iterator>::type
320lower_bound(CSR_Data<_value_type, _index_type>& data, const _arg_index_type& row, const _arg_index_type& column) {
321
322 // Check sizes
323 ASSERT_DEBUG(row >= 0, "Row index must be positive.");
324 ASSERT_DEBUG(column >= 0, "Columns index must be positive.");
325
326 const auto& const_iter = lower_bound(static_cast<const CSR_Data<_value_type, _index_type>&>(data), row, column);
327 return std::make_tuple(data.row_offset.begin() + std::distance(data.row_offset.cbegin(), std::get<0>(const_iter)),
328 data.i_column.begin() + std::distance(data.i_column.cbegin(), std::get<1>(const_iter)),
329 data.value.begin() + std::distance(data.value.cbegin(), std::get<2>(const_iter)));
330}
