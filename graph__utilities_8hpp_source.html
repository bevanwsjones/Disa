49 std::vector<std::size_t> level_traversal(const _graph& graph, const std::size_t i_start,
50  const std::size_t end_level = std::numeric_limits<std::size_t>::max()) {
51 
52  ASSERT_DEBUG(!graph.empty(), "Graph is empty.");
53  ASSERT_DEBUG(i_start < graph.size_vertex(),
54  "Starting vertex not in range (0, " + std::to_string(graph.size_vertex()) + "].");
55 
56  std::vector<std::size_t> vertex_level(graph.size_vertex(), std::numeric_limits<std::size_t>::max());

72 template<class _graph>
73 void level_traversal(const _graph& graph, std::queue<std::size_t>& vertex_queue, std::vector<std::size_t>& vertex_level,
74  const std::size_t end_level = std::numeric_limits<std::size_t>::max()) {
75 
76  ASSERT_DEBUG(!graph.empty(), "Graph is empty.");
77  ASSERT_DEBUG(vertex_level.size() == graph.size_vertex(), "Vertex level and graph size_vertex do not match.");
78  FOR_EACH_REF(level, vertex_level)++ level; // roll the vector over, before checking max.
79  ASSERT_DEBUG(*std::max_element(vertex_level.begin(), vertex_level.end()) < graph.size_vertex(),
80  "A vertex in vertex level not in graph range (0, " + std::to_string(graph.size_vertex()) + "].");
81 
82  while(!vertex_queue.empty()) {
83  const std::size_t front = vertex_queue.front();
84  vertex_queue.pop();
85  if(vertex_level[front] == end_level + 1) continue; // Do not add new vertices, +1 because of above increment.
86  FOR_EACH(vertex, graph[front]) {
87  if(!static_cast<bool>(vertex_level[vertex])) {
88  vertex_queue.push(vertex);
89  vertex_level[vertex] = vertex_level[front] + 1;
90  }
91  }
92  }
93  FOR_EACH_REF(level, vertex_level)-- level;
94 }

104 template<class _graph>
105 void level_expansion(const _graph& graph, const std::vector<std::size_t>& seeds,
106  std::vector<std::size_t>& vertex_color) {
107  ASSERT_DEBUG(!graph.empty(), "Graph is empty.");
108  ASSERT_DEBUG(!seeds.empty(), "There are no seeds to begin expansion.");
109  ASSERT_DEBUG(*std::max_element(seeds.begin(), seeds.end()) < graph.size_vertex(),
110  "A seed index is not in graph range [0, " + std::to_string(graph.size_vertex()) + ").");
111 
112  // Setup memory, and seed the queues and colors.
113  vertex_color.resize(graph.size_vertex());
114  std::vector<std::queue<std::size_t>> vertex_queues(seeds.size());
115  FOR_EACH_REF(color, vertex_color) color = std::numeric_limits<std::size_t>::max();
116  FOR(i_seed, seeds.size()) {
117  vertex_queues[i_seed].push({seeds[i_seed]});
118  vertex_color[seeds[i_seed]] = i_seed;
119  }
120 
121  // Color vertices.
122  std::size_t iteration = 0;
123  while(std::any_of(vertex_queues.begin(), vertex_queues.end(), [](const auto& queue) { return !queue.empty(); })) {
124  FOR(i_queue, vertex_queues.size()) {
125  // ensures we do a forward's and backwards sweep to try and keep expansion 'unbiased'.
126  auto& vertex_queue =
127  iteration % 2 == 0 ? vertex_queues[i_queue] : vertex_queues[vertex_queues.size() - i_queue - 1];
128 
129  // Perform a level expansion for this color.
130  std::queue<std::size_t> new_queue;
131  while(!vertex_queue.empty()) {
132  const std::size_t front = vertex_queue.front();
133  vertex_queue.pop();
134  FOR_EACH(vertex, graph[front]) {
135  if(vertex_color[vertex] == std::numeric_limits<std::size_t>::max()) {
136  new_queue.push(vertex);
137  vertex_color[vertex] = vertex_color[front];
138  }
139  }
140  }
141  std::swap(new_queue, vertex_queue);
142  }
143  ASSERT(iteration++ < graph.size_vertex(),
144  "Number of iterations have exceeded, " + std::to_string(iteration) + ". Is the graph disjoint?");
145  }
146 }

155 template<class _graph>
156 std::size_t pseudo_peripheral_vertex(const _graph& graph, std::size_t i_start = 0) {
157 
158  ASSERT_DEBUG(!graph.empty(), "The parsed graph is empty.");
159  ASSERT_DEBUG(i_start < graph.size_vertex(), "The parsed start vertex is not in the graph.");
160  if(graph.degree(i_start) == 0) {
161  WARNING("The parsed start vertex has a 0 degree.");
162  return i_start;
163  }
164 
165  // Calculate the distance of each vertex from the starting vertex
166  bool found = false;
167  std::size_t max_distance = 0;
168  std::size_t& pseudo_peripheral_node = i_start;
169  std::vector<std::size_t> distance;
170 
171  // Calculate the eccentricity of each vertex
172  while(!found) {
173  found = true;
174  distance = level_traversal(graph, pseudo_peripheral_node);
175  FOR(i_vertex, graph.size_vertex()) {
176  if(distance[i_vertex] > max_distance ||
177  (distance[i_vertex] == max_distance && graph.degree(i_vertex) < graph.degree(pseudo_peripheral_node))) {
178  max_distance = distance[i_vertex];
179  pseudo_peripheral_node = i_vertex;
180  found = false;
181  }
182  }
183  }
184  return pseudo_peripheral_node;
185 }

214 template<class _graph>
215 void eccentricity_vertex_breadth_first(const _graph& graph, const std::size_t i_start,
216  std::vector<std::size_t>& distance,
217  const std::size_t i_stop = std::numeric_limits<std::size_t>::max()) {
218  ASSERT_DEBUG(!graph.empty(), "The parsed graph is empty.");
219  ASSERT_DEBUG(i_start < graph.size_vertex(), "The parsed start vertex is not in the graph.");
220  ASSERT_DEBUG(i_start <= i_stop, "The parsed start vertex is greater than the parsed stop vertex.");
221  ASSERT_DEBUG(i_stop == std::numeric_limits<std::size_t>::max() || i_stop <= graph.size_vertex(),
222  "The stopping vertex is not in the graph size range [0, " + std::to_string(graph.size_vertex()) +
223  "] and not set to a default.");
224 
225  // Initialise the distance vector.
226  std::fill(distance.begin(), distance.end(), std::numeric_limits<std::size_t>::max());
227  distance.resize(graph.size_vertex() <= i_stop ? graph.size_vertex() : i_stop,
228  std::numeric_limits<std::size_t>::max());
229 
230  // Create a priority queue to store the vertices to be visited
231  std::queue<std::size_t> queue({i_start});
232  distance[i_start] = 0;
233 
234  // Visit each vertex in the graph
235  while(!queue.empty()) {
236  const std::size_t front = queue.front();
237  queue.pop();
238  if(front >= i_stop) continue; // we don't search.
239  FOR_EACH(vertex, graph[front]) {
240  if(distance[vertex] == std::numeric_limits<std::size_t>::max()) {
241  distance[vertex] = distance[front] + 1;
242  queue.push(vertex);
243  }
244  }
245  }
246 }
