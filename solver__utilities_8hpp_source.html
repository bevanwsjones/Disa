195 template<class _matrix, class _vector>
196 void Convergence_Data::update(const _matrix& coef, const _vector& solution, const _vector& constant) {
197 
198  std::tie(residual, residual_max) = compute_residual(coef, solution, constant);
199 
200  if(!iteration) {
201  residual_0 = residual;
202  residual_max_0 = residual_max;
203  }
204  residual_normalised = residual / residual_0;
205  residual_max_normalised = residual_max / residual_max_0;
206 
207  ++iteration;
208  duration = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::steady_clock::now() - start_time);
209 }

233 template<class _matrix, class _vector>
234 std::pair<Scalar, Scalar> compute_residual(const _matrix& coef, const _vector& solution, const _vector& constant) {
235 
236  // Check sizes
237  ASSERT_DEBUG(solution.size() != 0 && constant.size() != 0, "System size is 0.");
238  ASSERT_DEBUG(coef.size_row() == solution.size(),
239  "Coefficient matrix row size incompatible with solution vector size.");
240  ASSERT_DEBUG(coef.size_column() == constant.size(),
241  "Coefficient matrix column size incompatible with constant vector size.");
242 
243  Scalar l2_norm = 0.0;
244  Scalar linf_norm = 0.0;
245  for(auto i_row = 0; i_row < coef.size_row(); ++i_row) {
246  Scalar matrix_vector_row = 0;
247  for(auto column_iter = coef[i_row].cbegin(); column_iter != coef[i_row].cend(); ++column_iter)
248  matrix_vector_row += *column_iter * solution[column_iter.i_column()];
249 
250  // Compute the l2_norm and l_inf norm.
251  const Scalar& row_residual_squared = std::pow(matrix_vector_row - constant[i_row], 2);
252  l2_norm += row_residual_squared;
253  linf_norm = std::max(linf_norm, row_residual_squared);
254  }
255 
256  // 'Weight' the l_2 norm and return both norms.
257  return {std::sqrt(l2_norm / static_cast<Scalar>(solution.size())), std::sqrt(linf_norm)};
258 }
