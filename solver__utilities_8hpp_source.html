71 template<class _matrix, class _vector>
72 std::pair<Scalar, Scalar> compute_residual(const _matrix& coef, const _vector& solution, const _vector& constant) {
73 
74  // Check sizes
75  ASSERT_DEBUG(solution.size() != 0 && constant.size() != 0 , "System size is 0.");
76  ASSERT_DEBUG(coef.size_row() == solution.size(),
77  "Coefficient matrix row size incompatible with solution vector size.");
78  ASSERT_DEBUG(coef.size_column() == constant.size(),
79  "Coefficient matrix column size incompatible with constant vector size.");
80 
81  Scalar l2_norm = 0.0;
82  Scalar linf_norm = 0.0;
83  for(auto i_row = 0; i_row < coef.size_row(); ++i_row) {
84  Scalar matrix_vector_row = 0;
85  for(auto column_iter = coef[i_row].cbegin(); column_iter != coef[i_row].cend(); ++column_iter)
86  matrix_vector_row += *column_iter*solution[column_iter.i_column()];
87 
88  // Compute the l2_norm and l_inf norm.
89  const Scalar& row_residual_squared = std::pow(matrix_vector_row - constant[i_row], 2);
90  l2_norm += row_residual_squared;
91  linf_norm = std::max(linf_norm, row_residual_squared);
92  }
93 
94  // 'Weight' the l_2 norm and return both norms.
95  return {std::sqrt(l2_norm/static_cast<Scalar>(solution.size())), std::sqrt(linf_norm)};
96 }
