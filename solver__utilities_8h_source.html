158  [[nodiscard]] inline bool is_converged(const Convergence_Data& data) const {
159  if(data.iteration < min_iterations) return false;
160  if(data.iteration > max_iteration) return true;
161  if(data.residual_normalised > tolerance) return false;

163  return true;
