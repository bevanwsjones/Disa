574 template<typename _matrix = _matrix_type, typename std::enable_if_t<!std::is_const_v<_matrix>, bool> = true>
575 constexpr Scalar& operator[](const std::size_t& i_column) {
576 auto iter_element = matrix->lower_bound(row_index, i_column);
577 if(iter_element == end() || i_column != iter_element.i_column())
578 iter_element = matrix->insert(row_index, i_column, Scalar()).first;
579 return *iter_element;
580 };
581
589 constexpr const Scalar& operator[](const std::size_t& i_column) const {
590 ASSERT_DEBUG(matrix->contains(row_index, i_column),
591 "Trying to access a zero element at [" + matrix->row_column(row_index, i_column) + "].");
592 return *matrix->find(row_index, i_column);
593 };
