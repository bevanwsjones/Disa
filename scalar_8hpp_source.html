80[[nodiscard]] constexpr bool is_nearly_equal(const Scalar& scalar_0, const Scalar& scalar_1,
81 const Scalar& tolerance_relative = default_relative,
82 const Scalar& tolerance_absolute = default_absolute) {
83 ASSERT_DEBUG(scalar_epsilon <= tolerance_relative, "Relative tolerance " + std::to_string(tolerance_relative) +
84 " must be greater than scalar epsilon, " +
85 std::to_string(scalar_epsilon) + ".");
86 ASSERT_DEBUG(tolerance_relative <= static_cast<Scalar>(1),
87 "Relative tolerance " + std::to_string(tolerance_relative) + " greater than 1, will magnify the norm.");
88 if(scalar_0 == scalar_1) return true;
89 return std::abs(scalar_0 - scalar_1) <
90 std::max(tolerance_absolute,
91 tolerance_relative * std::min(std::abs(scalar_0) + std::abs(scalar_1), scalar_max));
92}

102[[nodiscard]] constexpr bool is_nearly_greater_equal(const Scalar& scalar_0, const Scalar& scalar_1,
103 const Scalar& tolerance_relative = default_relative,
104 const Scalar& tolerance_absolute = default_absolute) {
105 return scalar_0 > scalar_1 || is_nearly_equal(scalar_0, scalar_1, tolerance_relative, tolerance_absolute);
106}

116[[nodiscard]] constexpr bool is_nearly_less_equal(const Scalar& scalar_0, const Scalar& scalar_1,
117 const Scalar& tolerance_relative = default_relative,
118 const Scalar& tolerance_absolute = default_absolute) {
119 return scalar_0 < scalar_1 || is_nearly_equal(scalar_0, scalar_1, tolerance_relative, tolerance_absolute);

137[[nodiscard]] constexpr bool is_nearly_greater(const Scalar& scalar_0, const Scalar& scalar_1,
138 const Scalar& tolerance_relative = default_relative,
