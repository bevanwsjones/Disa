459  Matrix_Sparse(std::initializer_list<index_type> non_zero, std::initializer_list<index_type> index,
460  std::initializer_list<value_type> value, index_type column)
461  : row_offset(non_zero), column_index(index), element_value(value), column_size(column) {
462  ASSERT(row_offset.front() == 0, "First value must be zero, but is " + std::to_string(row_offset.front()) + ".");
463  ASSERT(row_offset.back() == column_index.size(), "Number of non-zeros does not match number of column non zeros");
464  ASSERT(std::is_sorted(row_offset.begin(), row_offset.end()), "Inconsistent non-zeros list, but be ascending.");
465  ASSERT(column_index.size() == value.size(), "Mis-match in column and value size, " +
466  std::to_string(column_index.size()) + " vs. " +
467  std::to_string(value.size()));
468 
469  std::vector<value_type> new_value;
470  std::vector<index_type> org_index;
471  FOR(row, row_offset.size() - 1) {
472  const index_type size_non_zero = row_offset[row + 1] - row_offset[row];
473  org_index.resize(size_non_zero);
474  new_value.resize(size_non_zero);
475  std::iota(org_index.begin(), org_index.end(), 0);
476  std::sort(org_index.begin(), org_index.end(), [&](const index_type index_0, const index_type index_1) {
477  return column_index[row_offset[row] + index_0] < column_index[row_offset[row] + index_1];
478  });
479 
480  FOR(i_non_zero, size_non_zero) {
481  new_value[i_non_zero] = element_value[row_offset[row] + org_index[i_non_zero]];
482  org_index[i_non_zero] = column_index[row_offset[row] + org_index[i_non_zero]];
483  ASSERT(org_index[i_non_zero] < column_size, "Column index, " + std::to_string(org_index[i_non_zero]) +
484  ", in row " + std::to_string(row) + " not in range" +
485  range_column() + ".");
486  }
487  ASSERT(std::adjacent_find(org_index.begin(), org_index.end()) == org_index.end(),
488  "Duplicate column index, " + std::to_string(column_size) + "in row " + std::to_string(row) + ".");
489  std::swap_ranges(new_value.begin(), new_value.end(),
490  std::next(element_value.begin(), static_cast<s_size_t>(row_offset[row])));
491  std::swap_ranges(org_index.begin(), org_index.end(),
492  std::next(column_index.begin(), static_cast<s_size_t>(row_offset[row])));
493  }
494  }

641  void resize(const_reference_index row, const_reference_index column) {
642 
643  using s_index_type = std::make_signed_t<index_type>;
644 
645  // resize rows first
646  if(row < size_row()) {
647  column_index.resize(row_offset[row]);
648  element_value.resize(row_offset[row]);
649  }
650  row_offset.resize(row + 1, row_offset.empty() ? 0 : row_offset.back());
651 
652  // resize columns
653  // Note: Since we may have to do column erases over multiple rows we do it directly here rather than use erase().
654  if(column < size_column()) {
655  index_type offset_loss = 0;
656  FOR(i_row, size_row()) {
657  const auto& iter_column_end =
658  column_index.begin() + static_cast<s_index_type>(row_offset[i_row + 1] - offset_loss);
659  auto iter_column = std::upper_bound(column_index.begin() + static_cast<s_index_type>(row_offset[i_row]),
660  iter_column_end, column - 1);
661  if(iter_column != iter_column_end) {
662  const s_index_type& start_distance = std::distance(column_index.begin(), iter_column);
663  const s_index_type& end_distance = std::distance(column_index.begin(), iter_column_end);
664  column_index.erase(iter_column, iter_column_end);
665  element_value.erase(element_value.begin() + start_distance, element_value.begin() + end_distance);
666  offset_loss += end_distance - start_distance;
667  }
668  row_offset[i_row + 1] -= offset_loss;
669  }
670  }
671  column_size = column;
672  }
