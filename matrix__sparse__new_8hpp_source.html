342  [[nodiscard]] operator const_reference_value() const {
343  auto iter = std::find(row_.i_column.begin(), row_.i_column.end(), i_column);
344  ASSERT_DEBUG(iter != row_.i_column.end(), "Attempting to implicitly insert an new element at (" +
345  std::to_string(row_.row()) + ", " + std::to_string(i_column) +
346  ") must use explicit assignment operator.");
347  return *(row_.value.begin() + std::distance(row_.i_column.begin(), iter));
348  }
349 
357  SparseAccessor& operator=(const_reference_value value) {
358  const index_type i_row = row_.row();
359  insert_or_assign(*row_.ptr_data, i_row, i_column, value);
360  row_ = std::move(Matrix_Sparse_Row(row_.ptr_data, row_.ptr_data->row_offset.begin() + i_row));
361  return *this;
362  }
363

365  Matrix_Sparse_Row& row_;
366  index_type i_column;
