326 class SparseAccessor {
327 public:
333 SparseAccessor(Matrix_Sparse_Row& row, index_type column) : row_(row), i_column(column) {}
334
342 [[nodiscard]] operator const_reference_value() const {
343 auto iter = std::find(row_.i_column.begin(), row_.i_column.end(), i_column);
344 ASSERT_DEBUG(iter != row_.i_column.end(), "Attempting to implicitly insert an new element at (" +
345 std::to_string(row_.row()) + ", " + std::to_string(i_column) +
346 ") must use explicit assignment operator.");
347 return *(row_.value.begin() + std::distance(row_.i_column.begin(), iter));
348 }
349
357 SparseAccessor& operator=(const_reference_value value) {
358 const index_type i_row = row_.row();
359 insert_or_assign(*row_.ptr_data, i_row, i_column, value);
360 row_ = std::move(Matrix_Sparse_Row(row_.ptr_data, row_.ptr_data->row_offset.begin() + i_row));
361 return *this;
362 }
363
364 private:
365 Matrix_Sparse_Row& row_;
366 index_type i_column;
367 };

455 [[nodiscard]] iterator end() noexcept { return iterator(&(*i_column.end()), &(*value.end())); }

462 return const_iterator(&(*i_column.begin()), &(*value.begin()));

464
469 [[nodiscard]] const_iterator end() const noexcept { return const_iterator(&(*i_column.end()), &(*value.end())); }
470
475 [[nodiscard]] const_iterator cbegin() const noexcept {
476 return const_iterator(&(*i_column.begin()), &(*value.begin()));
477 }
478
483 [[nodiscard]] const_iterator cend() const noexcept { return const_iterator(&(*i_column.end()), &(*value.end())); }
484
491 [[nodiscard]] const_reference_value at(const index_type& column) const {
492 auto iter = std::find(i_column.begin(), i_column.end(), column);
493 ASSERT(iter != i_column.end(),
494 "Column index " + std::to_string(column) + ", is out of range for row " + std::to_string(row()) + ".");
495 return *(value.begin() + std::distance(i_column.begin(), iter));
496 }
497
504 [[nodiscard]] reference_value at(const index_type& column) {
505 return const_cast<reference_value>(static_cast<const base_type*>(this)->at(column));
506 }
507
515 [[nodiscard]] const_reference_value operator[](const index_type& i_column) const {
516#ifdef DISA_DEBUG
517 return at(i_column);
518#else
519 auto iter = std::find(i_column.begin(), i_column.end(), i_column);
520 return *const_iterator(&(*iter), &(*(value.begin() + std::distance(i_column.begin(), iter))););
521#endif
522 }
523
531 SparseAccessor operator[](const index_type& i_column) noexcept { return SparseAccessor(*this, i_column); };
532
539 Matrix_Sparse_Row& operator=(Matrix_Sparse_Row&& other) noexcept {
540 if(this != &other) {
541 ptr_data = std::exchange(other.ptr_data, nullptr);
542 iter_row_offset = std::exchange(other.iter_row_offset, {});
543 i_column = std::exchange(other.i_column, {});
544 value = std::exchange(other.value, {});
545 }
546 return *this;
547 }
548
549 private:
550 csr_data* ptr_data;
551 index_iterator iter_row_offset;
552 std::span<index_type> i_column;
553 std::span<value_type> value;
554};
