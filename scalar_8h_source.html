75 [[nodiscard]] constexpr bool is_nearly_equal(const Scalar& scalar_0, const Scalar& scalar_1,
76  const Scalar& tolerance_relative = default_relative,
77  const Scalar& tolerance_absolute = default_absolute) {
78  ASSERT_DEBUG(scalar_epsilon <= tolerance_relative,
79  "Relative tolerance " + std::to_string(tolerance_relative) + " must be greater than scalar epsilon, "
80  + std::to_string(scalar_epsilon) + ".");
81  ASSERT_DEBUG(tolerance_relative <= static_cast<Scalar>(1),
82  "Relative tolerance " + std::to_string(tolerance_relative) + " greater than 1, will magnify the norm.");
83  if(scalar_0 == scalar_1) return true;
84  return std::abs(scalar_0 - scalar_1) <
85  std::max(tolerance_absolute, tolerance_relative*std::min(std::abs(scalar_0) + std::abs(scalar_1),
86  scalar_max));
87 }
