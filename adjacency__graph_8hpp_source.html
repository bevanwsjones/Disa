386template<bool _directed>
387Adjacency_Graph<_directed>::Adjacency_Graph(std::initializer_list<Edge> edge_graph) {
388 const auto iter = std::max_element(edge_graph.begin(), edge_graph.end(),
389 !_directed ?
390 [](const Edge& edge_0, const Edge& edge_1) {
391 return order_edge_vertex(&edge_0).second < order_edge_vertex(&edge_1).second;
392 } :
393 [](const Edge& edge_0, const Edge& edge_1) {
394 return edge_0.second < edge_1.second;
395 });
396 reserve(!_directed ? std::max(iter->first, iter->second) : iter->first, edge_graph.size());
397 FOR_EACH(edge, edge_graph) insert(edge);
398 shrink_to_fit();
399}

412template<bool _directed>
413bool Adjacency_Graph<_directed>::insert(const Edge& edge) {
414 ASSERT_DEBUG(edge.first != edge.second,
415 "Edge vertices identical, " + std::to_string(edge.first) + " and " + std::to_string(edge.second) + ".");
416
417 // Preliminaries check if the edge exists, then determine if we need to resize.
418 if(contains(edge)) return false;
419 const auto& [i_first_vertex, i_second_vertex] =
420 !_directed ? order_edge_vertex(&edge) : std::pair<const std::size_t&, const std::size_t&>({edge.first, edge.second});
421 if(std::max(i_first_vertex, i_second_vertex) >= size_vertex()) resize(std::max(i_first_vertex, i_second_vertex) + 1);
422
423 // Insert lower vertex.
424 insert_vertex_adjacent_list(i_first_vertex, i_second_vertex);
425 std::for_each(std::next(offset.begin(), static_cast<s_size_t>(i_first_vertex + 1)),
426 !_directed ? std::next(offset.begin(), static_cast<s_size_t>(i_second_vertex + 2)) : offset.end(),
427 [](std::size_t& off) { off++; });
428 if(_directed) return true;
429
430 // Insert upper vertex (tricky: last for loop is safe -> The highest vertex + 1 (for size) + 1 (for offset) <= end()).
431 insert_vertex_adjacent_list(i_second_vertex, i_first_vertex);
432 ++(*std::next(offset.begin(), static_cast<s_size_t>(i_second_vertex + 1)));
433 std::for_each(std::next(offset.begin(), static_cast<s_size_t>(i_second_vertex + 2)), offset.end(),
434 [](std::size_t& off) { off += 2; });
435
436 return true;
437}

445template<bool _directed>
446template<class _unary_predicate>
447void Adjacency_Graph<_directed>::erase_if(_unary_predicate delete_vertex) {
448
449 std::size_t removed = 0;
450 std::size_t i_vertex = 0;
451 std::vector<bool> adjacency_delete(vertex_adjacent_list.size(), false);
452 std::vector<std::size_t> new_indexes(vertex_adjacent_list.size(), std::numeric_limits<std::size_t>::max());
453
454 // Loop over the data, and determine the new offsets and which entries in vertex_adjacent_list need to be removed.
455 FOR(i_vertex_old, size_vertex()) {
456 const auto& adjacency = vertex_adjacency_iter(i_vertex_old);
457 const auto& begin_delete = adjacency_delete.begin() + std::distance(vertex_adjacent_list.begin(), adjacency.first);
458 const auto& end_delete = adjacency_delete.begin() + std::distance(vertex_adjacent_list.begin(), adjacency.second);
459
460 offset[i_vertex_old] -= removed;
461
462 // Only remove vertices which will be taken out, and record new indexes.
463 if(!delete_vertex(i_vertex_old)) {
464 new_indexes[i_vertex_old] = i_vertex++;
465 removed += std::count_if(adjacency.first, adjacency.second, delete_vertex);
466 std::transform(adjacency.first, adjacency.second, begin_delete, delete_vertex);
467 } else { // All adjacency entries need to be removed, since the vertex is to be removed.
468 removed += std::distance(adjacency.first, adjacency.second);
469 std::transform(begin_delete, end_delete, begin_delete, [](const auto) { return true; });
470 }
471 }
472 offset.back() -= removed;
473
474 // Erase offset entries for the removed vertices.
475 i_vertex = 0;
476 offset.erase(std::remove_if(offset.begin(), offset.end(),
477 [&](const auto) { return i_vertex < size_vertex() ? delete_vertex(i_vertex++) : false; }),
478 offset.end());
479
480 // Erase removed vertices from the adjacency lists, and then relabel.
481 i_vertex = 0;
482 vertex_adjacent_list.erase(std::remove_if(vertex_adjacent_list.begin(), vertex_adjacent_list.end(),
483 [&](const auto) { return adjacency_delete[i_vertex++]; }),
484 vertex_adjacent_list.end());
485
486 std::transform(vertex_adjacent_list.begin(), vertex_adjacent_list.end(), vertex_adjacent_list.begin(),
487 [&](const auto& i_old_vertex) { return new_indexes[i_old_vertex]; });
488
489 ASSERT(offset.back() == vertex_adjacent_list.size(),
490 "Total offsets no longer match vertex size while reducing to subgraph, " + std::to_string(offset.back()) +
491 " vs. " + std::to_string(vertex_adjacent_list.size()) + ".");
492};

500template<bool _directed>
501void Adjacency_Graph<_directed>::resize(const std::size_t& size) {
502
503 // Branch based on expansion or contraction of the graph.
504 if(size_vertex() < size) {
505 offset.resize(size + 1, offset.empty() ? 0 : offset.back());
506 } else {
507 // Resize the containers.
508 offset.resize(size + 1);
509 vertex_adjacent_list.resize(offset.back());
510
511 // Loop over remaining vertices, looking for edges which no longer exist.
512 std::size_t offset_carry = 0;
513 FOR(i_vertex, size_vertex()) {
514 auto [adjacency_begin, adjacency_end] = vertex_adjacency_iter(i_vertex);
515 const auto erase_iter = std::lower_bound(adjacency_begin, adjacency_end, size);
516 if(erase_iter != adjacency_end) {
517 const auto offset_current = std::distance(erase_iter, adjacency_end);
518 vertex_adjacent_list.erase(erase_iter, adjacency_end);
519 offset[i_vertex + 1] -= offset_current;
520 offset_carry += offset_current;
521 }
522 offset[i_vertex + 2] -= offset_carry;
523 }
524 }
525}

535template<bool _directed>
536[[nodiscard]] bool Adjacency_Graph<_directed>::contains(const Edge& edge) const {
537 ASSERT_DEBUG(edge.first != edge.second,
538 "Edge vertices identical, " + std::to_string(edge.first) + " and " + std::to_string(edge.second) + ".");
539
540 if(empty()) return false;
541 if(std::max(edge.first, edge.second) >= size_vertex()) return false;
542 auto iter = std::lower_bound((*this)[edge.first].begin(), (*this)[edge.first].end(), edge.second);
543 return iter != (*this)[edge.first].end() && *iter == edge.second;
544}

558template<bool _directed>
559Adjacency_Graph<_directed> Adjacency_Graph<_directed>::reorder(const std::vector<std::size_t>& permutation) {
560
561 ASSERT_DEBUG(permutation.size() == size_vertex(), "Incorrect sizes, " + std::to_string(permutation.size()) + " vs. " +
562 std::to_string(size_vertex()) + ".");
563 ASSERT_DEBUG(
564 std::accumulate(permutation.begin(), permutation.end(), 0.0) == (size_vertex() - 1) * size_vertex() / 2.0,
565 "Checksum for parsed re_ordering failed, are the elements unique?");
566
567 // Allocate memory.
568 Adjacency_Graph graph;
569 graph.offset.resize(offset.size());
570 graph.vertex_adjacent_list.resize(vertex_adjacent_list.size());
571
572 // Set up the offsets for the new graph.
573 if(!empty()) {
574 graph.offset[0] = 0;
575 FOR(i_old, size_vertex()) graph.offset[permutation[i_old] + 1] = offset[i_old + 1] - offset[i_old];
576 FOR(i_new, size_vertex()) graph.offset[i_new + 1] += graph.offset[i_new];
577 }
578
579 // Populate the new graph using the reorder mapping.
580 FOR(i_old, size_vertex()) {
581 const std::size_t& i_new = permutation[i_old];
582 auto adjacency_iter = vertex_adjacency_iter(i_old);
583 std::transform(adjacency_iter.first, adjacency_iter.second,
584 std::next(graph.vertex_adjacent_list.begin(), static_cast<s_size_t>(graph.offset[i_new])),
585 [permutation](const std::size_t& i_old) { return permutation[i_old]; });
586 adjacency_iter = graph.vertex_adjacency_iter(i_new);
587 std::sort(adjacency_iter.first, adjacency_iter.second);
588 }
589
590 // swap class data and return.
591 this->swap(graph);
592 return graph;
593}

603template<bool _directed>
604void Adjacency_Graph<_directed>::insert_vertex_adjacent_list(std::size_t vertex, std::size_t insert_vertex) {
605 const auto& adjacency = vertex_adjacency_iter(vertex);
606 const auto& insert_iter = std::lower_bound(adjacency.first, adjacency.second, insert_vertex);
607 const auto& distance = std::distance(vertex_adjacent_list.begin(), insert_iter);
608 vertex_adjacent_list.insert(vertex_adjacent_list.begin() + distance, insert_vertex);
609}

627template<bool _directed>
628std::ostream& operator<<(std::ostream& ostream, const Adjacency_Graph<_directed>& graph) {
629 FOR(i_vertex, graph.size_vertex()) {
630 if(i_vertex != 0) ostream << "\n";
631 FOR_EACH(adjacent_vertex, graph[i_vertex])
632 ostream << adjacent_vertex << (adjacent_vertex != graph[i_vertex].back() ? ", " : "");
633 if(graph[i_vertex].empty()) ostream << ".";
634 }
635 return ostream;
