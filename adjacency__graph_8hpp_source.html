35 template<bool _directed>
36 Adjacency_Graph<_directed>::Adjacency_Graph(std::initializer_list<Edge> edge_graph) {
37  const auto iter = std::max_element(edge_graph.begin(), edge_graph.end(),
38  !_directed ?
39  [](const Edge& edge_0, const Edge& edge_1) {
40  return order_edge_vertex(&edge_0).second < order_edge_vertex(&edge_1).second;
41  } :
42  [](const Edge& edge_0, const Edge& edge_1) {
43  return edge_0.second < edge_1.second;
44  });
45  reserve(!_directed ? std::max(iter->first, iter->second) : iter->first, edge_graph.size());
46  FOR_EACH(edge, edge_graph) insert(edge);
47  shrink_to_fit();
48 }

61 template<bool _directed>
62 bool Adjacency_Graph<_directed>::insert(const Edge& edge) {
63  ASSERT_DEBUG(edge.first != edge.second, "Edge vertices identical, " + std::to_string(edge.first) + " and "
64  + std::to_string(edge.second) + ".");
65 
66  // Preliminaries check if the edge exists, then determine if we need to resize.
67  if(contains(edge)) return false;
68  const auto& [i_first_vertex, i_second_vertex] =
69  !_directed ? order_edge_vertex(&edge) : std::pair<const std::size_t&, const std::size_t&>({edge.first, edge.second});
70  if(std::max(i_first_vertex, i_second_vertex) >= size_vertex()) resize(std::max(i_first_vertex, i_second_vertex) + 1);
71 
72  // Insert lower vertex.
73  insert_vertex_adjacent_list(i_first_vertex, i_second_vertex);
74  std::for_each(std::next(offset.begin(), static_cast<s_size_t>(i_first_vertex + 1)),
75  !_directed ? std::next(offset.begin(), static_cast<s_size_t>(i_second_vertex + 2)) : offset.end(),
76  [](std::size_t& off){off++;});
77  if(_directed) return true;
78 
79  // Insert upper vertex (tricky: last for loop is safe -> The highest vertex + 1 (for size) + 1 (for offset) <= end()).
80  insert_vertex_adjacent_list(i_second_vertex, i_first_vertex);
81  ++(*std::next(offset.begin(), static_cast<s_size_t>(i_second_vertex + 1)));
82  std::for_each(std::next(offset.begin(), static_cast<s_size_t>(i_second_vertex + 2)), offset.end(),
83  [](std::size_t& off){off += 2;});
84 
85 
86 
87  return true;
88 }

97 template<class _unary_predicate>
98 void Adjacency_Graph<_directed>::erase_if(_unary_predicate delete_vertex) {
99 
100  std::size_t removed = 0;
101  std::size_t i_vertex = 0;
102  std::vector<bool> adjacency_delete(vertex_adjacent_list.size(), false);
103  std::vector<std::size_t> new_indexes(vertex_adjacent_list.size(), std::numeric_limits<std::size_t>::max());
104 
105  // Loop over the data, and determine the new offsets and which entries in vertex_adjacent_list need to be removed.
106  FOR(i_vertex_old, size_vertex()) {
107  const auto& adjacency = vertex_adjacency_iter(i_vertex_old);
108  const auto& begin_delete = adjacency_delete.begin() + std::distance(vertex_adjacent_list.begin(), adjacency.first);
109  const auto& end_delete = adjacency_delete.begin() + std::distance(vertex_adjacent_list.begin(), adjacency.second);
110 
111  offset[i_vertex_old] -= removed;
112 
113  // Only remove vertices which will be taken out, and record new indexes.
114  if(!delete_vertex(i_vertex_old)) {
115  new_indexes[i_vertex_old] = i_vertex++;
116  removed += std::count_if(adjacency.first, adjacency.second, delete_vertex);
117  std::transform(adjacency.first, adjacency.second, begin_delete, delete_vertex);
118  } else { // All adjacency entries need to be removed, since the vertex is to be removed.
119  removed += std::distance(adjacency.first, adjacency.second);
120  std::transform(begin_delete, end_delete, begin_delete, [](const auto) { return true; });
121  }
122  }
123  offset.back() -= removed;
124 
125  // Erase offset entries for the removed vertices.
126  i_vertex = 0;
127  offset.erase(std::remove_if(offset.begin(), offset.end(),
128  [&](const auto) { return i_vertex < size_vertex() ? delete_vertex(i_vertex++) : false; }),
129  offset.end());
130 
131  // Erase removed vertices from the adjacency lists, and then relabel.
132  i_vertex = 0;
133  vertex_adjacent_list.erase(std::remove_if(vertex_adjacent_list.begin(), vertex_adjacent_list.end(),
134  [&](const auto) { return adjacency_delete[i_vertex++]; }),
135  vertex_adjacent_list.end());
136 
137  std::transform(vertex_adjacent_list.begin(), vertex_adjacent_list.end(), vertex_adjacent_list.begin(),
138  [&](const auto& i_old_vertex) { return new_indexes[i_old_vertex]; });
139 
140  ASSERT(offset.back() == vertex_adjacent_list.size(),
141  "Total offsets no longer match vertex size while reducing to subgraph, " + std::to_string(offset.back())
142  + " vs. " + std::to_string(vertex_adjacent_list.size()) + ".");
143 };

151 template<bool _directed>
152 void Adjacency_Graph<_directed>::resize(const std::size_t& size) {
153 
154  // Branch based on expansion or contraction of the graph.
155  if(size_vertex() < size) {
156  offset.resize(size + 1, offset.empty() ? 0 : offset.back());
157  }
158  else {
159  // Resize the containers.
160  offset.resize(size + 1);
161  vertex_adjacent_list.resize(offset.back());
162 
163  // Loop over remaining vertices, looking for edges which no longer exist.
164  std::size_t offset_carry = 0;
165  FOR(i_vertex, size_vertex()) {
166  auto [adjacency_begin, adjacency_end] = vertex_adjacency_iter(i_vertex);
167  const auto erase_iter = std::lower_bound(adjacency_begin, adjacency_end, size);
168  if(erase_iter != adjacency_end) {
169  const auto offset_current = std::distance(erase_iter, adjacency_end);
170  vertex_adjacent_list.erase(erase_iter, adjacency_end);
171  offset[i_vertex + 1] -= offset_current;
172  offset_carry += offset_current;
173  }
174  offset[i_vertex + 2] -= offset_carry;
175  }
176  }
177 }

187 template<bool _directed>
188 [[nodiscard]] bool Adjacency_Graph<_directed>::contains(const Edge& edge) const {
189  ASSERT_DEBUG(edge.first != edge.second, "Edge vertices identical, " + std::to_string(edge.first) + " and "
190  + std::to_string(edge.second) + ".");
191 
192  if(empty()) return false;
193  if(std::max(edge.first, edge.second) >= size_vertex()) return false;
194  auto iter = std::lower_bound((*this)[edge.first].begin(), (*this)[edge.first].end(), edge.second);
195  return iter != (*this)[edge.first].end() && *iter == edge.second;
196 }

210 template<bool _directed>
211 Adjacency_Graph<_directed> Adjacency_Graph<_directed>::reorder(const std::vector<std::size_t>& permutation) {
212 
213  ASSERT_DEBUG(permutation.size() == size_vertex(), "Incorrect sizes, " + std::to_string(permutation.size()) + " vs. "
214  + std::to_string(size_vertex()) + ".");
215  ASSERT_DEBUG(std::accumulate(permutation.begin(), permutation.end(), 0.0) == (size_vertex() - 1)*size_vertex()/2.0,
216  "Checksum for parsed re_ordering failed, are the elements unique?");
217 
218  // Allocate memory.
219  Adjacency_Graph graph;
220  graph.offset.resize(offset.size());
221  graph.vertex_adjacent_list.resize(vertex_adjacent_list.size());
222 
223  // Set up the offsets for the new graph.
224  if(!empty()) {
225  graph.offset[0] = 0;
226  FOR(i_old, size_vertex()) graph.offset[permutation[i_old] + 1] = offset[i_old + 1] - offset[i_old];
227  FOR(i_new, size_vertex()) graph.offset[i_new + 1] += graph.offset[i_new];
228  }
229 
230  // Populate the new graph using the reorder mapping.
231  FOR(i_old, size_vertex()) {
232  const std::size_t& i_new = permutation[i_old];
233  auto adjacency_iter = vertex_adjacency_iter(i_old);
234  std::transform(adjacency_iter.first, adjacency_iter.second,
235  std::next(graph.vertex_adjacent_list.begin(), static_cast<s_size_t>(graph.offset[i_new])),
236  [permutation](const std::size_t& i_old){return permutation[i_old];});
237  adjacency_iter = graph.vertex_adjacency_iter(i_new);
238  std::sort(adjacency_iter.first, adjacency_iter.second);
239  }
240 
241  // swap class data and return.
242  this->swap(graph);
243  return graph;
244 }

254 template<bool _directed>
255 void Adjacency_Graph<_directed>::insert_vertex_adjacent_list(std::size_t vertex, std::size_t insert_vertex) {
256  const auto& adjacency = vertex_adjacency_iter(vertex);
257  const auto& insert_iter = std::lower_bound(adjacency.first, adjacency.second, insert_vertex);
258  const auto& distance = std::distance(vertex_adjacent_list.begin(), insert_iter);
259  vertex_adjacent_list.insert(vertex_adjacent_list.begin() + distance, insert_vertex);
260 }
