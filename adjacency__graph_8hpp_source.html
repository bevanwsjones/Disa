37 void Adjacency_Graph::erase_if(_unary_predicate delete_vertex) {
38 
39  std::size_t removed = 0;
40  std::size_t i_vertex = 0;
41  std::vector<bool> adjacency_delete(vertex_adjacent_list.size(), false);
42  std::vector<std::size_t> new_indexes(vertex_adjacent_list.size(), std::numeric_limits<std::size_t>::max());
43 
44  // Loop over the data, and determine the new offsets and which entries in vertex_adjacent_list need to be removed.
45  FOR(i_vertex_old, size_vertex()) {
46  const auto& adjacency = vertex_adjacency_iter(i_vertex_old);
47  const auto& begin_delete = adjacency_delete.begin() + std::distance(vertex_adjacent_list.begin(), adjacency.first);
48  const auto& end_delete = adjacency_delete.begin() + std::distance(vertex_adjacent_list.begin(), adjacency.second);
49 
50  offset[i_vertex_old] -= removed;
51 
52  // Only remove vertices which will be taken out, and record new indexes.
53  if(!delete_vertex(i_vertex_old)) {
54  new_indexes[i_vertex_old] = i_vertex++;
55  removed += std::count_if(adjacency.first, adjacency.second, delete_vertex);
56  std::transform(adjacency.first, adjacency.second, begin_delete, delete_vertex);
57  } else { // All adjacency entries need to be removed, since the vertex is to be removed.
58  removed += std::distance(adjacency.first, adjacency.second);
59  std::transform(begin_delete, end_delete, begin_delete, [](const auto) { return true; });
60  }
61  }
62  offset.back() -= removed;
63 
64  // Erase offset entries for the removed vertices.
65  i_vertex = 0;
66  offset.erase(std::remove_if(offset.begin(), offset.end(),
67  [&](const auto) { return i_vertex < size_vertex() ? delete_vertex(i_vertex++) : false; }),
68  offset.end());
69 
70  // Erase removed vertices from the adjacency lists, and then relabel.
71  i_vertex = 0;
72  vertex_adjacent_list.erase(std::remove_if(vertex_adjacent_list.begin(), vertex_adjacent_list.end(),
73  [&](const auto) { return adjacency_delete[i_vertex++]; }),
74  vertex_adjacent_list.end());
75  std::transform(vertex_adjacent_list.begin(), vertex_adjacent_list.end(), vertex_adjacent_list.begin(),
76  [&](const auto& i_old_vertex) { return new_indexes[i_old_vertex]; });
77 
78  ASSERT(offset.back() == vertex_adjacent_list.size(),
79  "Total offsets no longer match vertex size while reducing to subgraph, " + std::to_string(offset.back())
80  + " vs. " + std::to_string(vertex_adjacent_list.size()) + ".");
81 };
