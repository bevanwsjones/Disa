53inline std::basic_string<char> console_format(const Log_Level level, const std::source_location& location) {
54 auto const iter = std::basic_string_view<char>(location.file_name()).find_last_of('/');
55 const auto suffix =
56 std::basic_string<char>(location.file_name()).substr(iter + 1) + "::" + std::to_string(location.line()) + "]: ";
57 switch(level) {
58 case Log_Level::Debug:
59 return "[Debug|" + std::basic_string(suffix);
60 case Log_Level::Info:
61 return "[Info |" + std::basic_string(suffix);
62 case Log_Level::Warning:
63 return "[Warn |" + std::basic_string(suffix);
64 case Log_Level::Error:
65 return "[Error|" + std::basic_string(suffix);
66 default:
67 std::cout << "[Error|" + std::basic_string(suffix) + " Unrecognised log level parsed.";
68 exit(1);
69 }
70} //console_format
